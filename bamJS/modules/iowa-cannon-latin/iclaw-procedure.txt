INSERT into iclaw_authors (abbreviation,
author_latin,
dump_id,
author_english,
modern_geographic_identity,
ancient_geographic_identity,
pleiades_id,
century,
date_of_birth,
date_of_death,
floruit,
late_antique,
christian)
(
select
abbreviation,
author_latin,
min (dump_id) as dump_id,
min (author_english) as author_english, 
min (modern_geographic_identity) as modern_geographic_identity,
min (ancient_geographic_identity) as ancient_geographic_identity,
min (pleiades_id) as pleiades_id,
min (century) as century,
min (date_of_birth) as date_of_birth,
min (date_of_death) as date_of_death,
min (floruit) as floruit,
min (late_antique) as late_antique,
min (christian) as christian
from tbib_dump
GROUP BY abbreviation,
author_latin
ORDER BY author_latin
);

ICLAW SQL
The purpose here is to join any singular ICLAW entry with a pleiades ID. THis will not capture multiple entries, which should be solved when the database is changed to a relational table.

SELECT
iclaw_authors.bam_id,
iclaw_authors.abbreviation,
iclaw_authors.author_latin,
iclaw_authors.author_english,
iclaw_authors.modern_geographic_identity,
iclaw_authors.ancient_geographic_identity,
iclaw_authors.pleiades_id,
iclaw_authors.century,
iclaw_authors.date_of_birth,
iclaw_authors.date_of_death,
iclaw_authors.floruit,
iclaw_authors.late_antique,
iclaw_authors.christian,
pplaces.reprlat,
pplaces.reprlong,
pplaces.title,
pplaces.id AS pid
FROM
iclaw_authors LEFT JOIN pplaces ON iclaw_authors.pleiades_id = pplaces.id
